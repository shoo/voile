static struct Point
{
	@name("xValue")
	int x;
	@name("yValue")
	int y = 10;
}
static struct Data
{
	string key;
	int    value;
	@ignore    int    testval;
	@essential Point  pt;
	Point[] points;
	Point[string] pointMap;
	
	@converter!SysTime(jv=>SysTime.fromISOExtString(jv.str),
	                   v =>JSONValue(v.toISOExtString()))
	SysTime time;
}
Data x, y, z;
x.key = "xxx";
x.value = 200;
x.pt.x = 300;
x.pt.y = 400;
x.testval = 100;

x.time = tim;
