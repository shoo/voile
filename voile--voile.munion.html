パターンマッチ
Examples: 
import std.exception;
import std.conv: ConvException;
import voile.misc: nogcEnforce;
alias U = TypeEnum!(int, short, long);
U dat;

// マッチ関数を呼び出し
dat.set!0 = 1;
assert(dat.match!(
	(int x)  => x + 1, /* call */
	()       => 50,
	(long x) => x + 100) == 2);

// デフォルトを設定できる(empty || どれにもヒットしない)
dat.clear();
assert(dat.match!(
	(int x)  => x + 1,
	()       => 50, /* call */
	(long x) => x + 100) == 50);
// 例外をキャッチできる
dat = cast(long)5;
assert(dat.match!(
	(Exception e) => 100, /* call 2 */
	other         => 50,
	(long x)      => nogcEnforce(0)/* call 1 */) == 100);
